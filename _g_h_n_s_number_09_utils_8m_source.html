00040 - (NSString *)gh_humanSizeWithDelimiter:(NSString *)delimiter {
00041   double value = [self doubleValue];
00042     double byteTest = 1024;
00043     double kilobyteTest = 1024 * byteTest;
00044     double megaByteTest = 1024 * kilobyteTest;
00045   
00046   if (value < byteTest) {
00047     return [NSString stringWithFormat:@"%.0f%@B", value, delimiter];
00048   } else if (value < kilobyteTest) {
00049     double d = value / byteTest;
00050     return [NSString stringWithFormat:@"%.0f%@KB", d, delimiter];
00051   } else if (value < megaByteTest) {
00052     double d = value / kilobyteTest;        
00053     return [NSString stringWithFormat:@"%.1f%@MB", d, delimiter];
00054   } else {
00055         double d = value / megaByteTest;
00056         return [NSString stringWithFormat:@"%.2f%@GB", d, delimiter];
00057   }
00058   return nil;
00059 }

00069 + (NSString *)gh_ordinalizeEn:(NSInteger)value {
00070   NSString *suffix = nil;
00071   if (value > 0) {
00072     switch(value % 10) {
00073       case 1: suffix = @"st"; break;
00074       case 2: suffix = @"nd"; break;
00075       case 3: suffix = @"rd"; break;
00076       default: suffix = @"th"; break;
00077     }
00078     if (value % 100 >= 11 && value % 100 <= 13) suffix = @"th"; // Handle 11-13
00079     if (suffix) return [NSString stringWithFormat:@"%d%@", value, suffix];
00080   }
00081   return [NSString stringWithFormat:@"%d", value];
00082 }
